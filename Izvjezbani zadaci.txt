#include <stdio.h>
#include <math.h>
#include <stdlib.h>
#include <time.h>

int duzina(const char *a){
    int brojac=0;
  
  for(int i=0; a[i]!='\0';i++){
      brojac++;
  }
    
    return brojac;
}


int jesuLiJednaki(const char *a, const char*b, int n){
    
    if(duzina(a)<n || duzina(b) < n){
        return 0;
    }
    
    for(int i=0; i<n; i++){
        if(a[i] != b[i]){
            return 0;
        }
    }
    return 1;
}


int main()
{
    
    char a[26];
    
    fgets(a, sizeof(a), stdin);
    
    int len=duzina(a);
    
    if(a[len-1] == '\n'){
        a[len-1] == '\0';
    }
    
    int rezultat1 = jesuLiJednaki(a, "Moram da shvatim!", 2);
    printf("rezultat %d\n", rezultat1);
    int rezultat2 = jesuLiJednaki(a, "Moram da shvatim!",5);
    printf("rezultat %d", rezultat2);
    
    return 0;
}

int odredivanjeBrojaDijelitelja(int x){
    int brojac=0;
    for(int i=1; i<=x; i++){
        if(x%i == 0){
            brojac++;
        }
    }
    return brojac;
}

int najveciBroj(int n, int m){
    int max=0;
    int broj = odredivanjeBrojaDijelitelja(m);
    
    for(int i=n ; i<=m; i++){
        int trenutniBroj = odredivanjeBrojaDijelitelja(i);
        if(trenutniBroj > broj ){
            max = i;
           broj = trenutniBroj;
        }
    }
    return max;
}

int main()
{
    int donja, gornja;
    
    do{
        scanf("%d", &gornja);
        scanf("%d", &donja);
    }while(gornja<donja);
    
    int rezultat = najveciBroj(donja,gornja);
    
    printf("%d", rezultat);
    
    return 0;
}

int jeLiProst(int broj){
    if(broj<2){
        return 0;
    }
    
    for(int i=2; i*i <= broj; i++){
        if(broj%i==0) return 0;
    }
    return 1;
}

int brojProstihBrojeva(const char *a){
    int count =0;
    
    for(int i=0; a[i]!='\0'; i++){
        int ascii = (int)a[i];
        if(jeLiProst(ascii)){
            count++;
        }
    }
    
    return count;
}

#include <stdio.h>


double art(double *a){
    double sum =0;
    
    for(int i=0; i<15; i++){
        sum += a[i];
    }
    
    if(sum == 0){
        return 0;
    }
    
    return sum/15;
}


double kvocijent(double *a){
    int brV=0, brM =0;
    double aritmeticka = art(a);
    
    for(int i=0; i<15;i++){
        if(a[i]>aritmeticka){
            brV++;
        }else{
            brM++;
        }
    }
    
    if(brM == 0){
        return 0;
    }
    return (double)brV/brM;
}


int main()
{
    double a[15];
    
    for(int i=0; i<15; i++){
        scanf("%lf", &a[i]);
    }
    
    
    double rezultat = kvocijent(a);
    
    printf("%.2f", rezultat);
    return 0;
}

#include <stdio.h>

int najvecaZnamenka(int n){
    
    int max=0;
    
    if(n==0){
        return 0;
    }
    while(n>0){
        int x = n%10;
        if(x>max){
            max = x;
        }
        
        n /=10;
    }
    return max;
}

int najmanjaZnamenka(int n){
    int min=9;
    
    while(n>0){
        int x=n%10;
        if(x<min){
            min =x;
        }
        n /=10;
    }
    return min;
}


int main()
{
    
    int n;
    int max, min;
    
    do{
        scanf("%d", &n);
        max=najvecaZnamenka(n);
        min=najmanjaZnamenka(n);
    }while(min*min != max);
    
    printf("%d\n", max);
    printf("%d", min);
    return 0;
}
